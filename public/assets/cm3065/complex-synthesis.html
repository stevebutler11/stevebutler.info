Reminder: simple synthesis¶

# adapted from thinkdsp p65
def synthesize_simple(amps, fs, ts):
    args = np.outer(ts, fs) # 
    M = np.cos(np.pi*2 * args)
    ys = np.dot(M, amps)
    return ys

amps = np.array([0.5, 0.25, 0.125])
freqs = np.array([1, 2, 3])
times = np.linspace(0, 1, 100)
signal = synthesize_simple(amps, freqs, times)

plt.plot(signal)
plt.title("Output of simple synthesis")

# from thinkdsp p80
def synthesize_complex(amps, fs, ts):
    args = np.outer(ts, fs)
    M = np.exp(1j * (np.pi*2 * args)) # 1j converts to complex 
    print("M contains...", type(M[0][0]))
    ys = np.dot(M, amps)
    return ys

    
amps = np.array([0.5, 0.5, 0.5])
freqs = np.array([1, 2, 3])
times = np.linspace(0, 1, 100)
signal = synthesize_complex(amps, freqs, times)

# notice we pull out real and imaginary parts
plt.plot(signal.real, linestyle="dashed", label="real")
plt.plot(signal.imag, label="imag")

plt.legend(loc="lower left")
plt.title("Output of complex sytnthesis")
